#include <iostream>
#include <bitset>

#define BITSPERWORD 32
#define SHIFT 5
#define maska 0x1F
#define N 10000000

using namespace std;

int a[1+N/BITSPERWORD]; // 31251

void set(int i) {
	a[i>>SHIFT] |= (1<<(i&maska));
	const int n = 8;
	cout << "debug::i " << std::bitset<n>(i).to_string() << endl;
	cout << "debug::i>>SHIFT " << std::bitset<n>(i>>SHIFT).to_string() << endl;
}

int main() {
//	for(int i(0); i < N; i++) {
//		a[0] = 0;
//	}


	set(35);
	set(79);

	set(96); // +32
	set(128); // +32
	for(int i(0); i < 1000; i++) {
		cout << "[" << i << "]" << "{" << std::bitset<8>(a[i]).to_string() << "} " << a[i] << ", ";
	}
	return 0;
}
